# Checkout the master branch
git checkout master

# Pull the latest changes to ensure you are up to date
git pull origin master

# Merge the sell-options branch
git merge sell-options

# Resolve any conflicts, then:
git add <conflicted-files>
git commit

# Push the changes to remote
git push origin master

# Delete the local sell-options branch
git branch -d sell-options

# Delete the remote sell-options branch
git push origin --delete sell-options
